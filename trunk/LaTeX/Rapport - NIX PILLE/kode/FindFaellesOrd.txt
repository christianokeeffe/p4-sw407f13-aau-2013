private string FindFaellesOrd (string A, string B)
{
  string a = A.ToLower();
  string b = B.ToLower();
  string EndeligeString = "";
  //Løkken tjekker bogstaverne i de to strenge
  for (int i = 0; i < a.Length; i++){
    for (int j= 0; j < b.Length; j++){
      //Når to bogstaver matcher tælles der frem efter for at tjekke om de efterfølgende bogstaver passer sammen
      if(a[i] == b[j] && a[i] != ' '){
        int q = 0;
        //Den kører indtil en af strengene ikke er længere eller de ikke længere matcher
        while (j+q < b.Length && i+q < a.Length && a[i+q] == b[j+q])
        {
          q++;
        }
        //Et tjek, før den bliver gemt, på om alt stemmer over ens, ordet har den rigtige længde og at det også er i begge start strenge
        if (EndeligeString.Length <= q && Global.TjekForStrengDelAfOrd(A, B.Substring(j, q)) && Global.TjekForStrengDelAfOrd(B, B.Substring(j, q)))
        {
          EndeligeString = B.Substring(j,q);
        }
      }
    }
  }

  return EndeligeString.Trim();
}
