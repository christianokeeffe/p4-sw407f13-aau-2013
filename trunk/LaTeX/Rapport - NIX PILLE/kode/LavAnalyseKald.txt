public List<Tilbud> LavAnalyse(int BrugerID)
{
  //Henter alle brugerens vare liggende på deres tilknyttede lister.
  List<ListElementInfo> AnalyseList = HentBrugerListe(BrugerID);
  List<Tilbud> returliste = null;

  if (AnalyseList == null)
  {
    return null;
  }
  else
  {
    //Hent alle nøgleord ud fra vare
    AnalyseList = HentNoegleOrd(AnalyseList);
    //Tjek nøgleord for stavefejl mm.
    AnalyseList = TjekNoegleOrdForFejl(AnalyseList);
    //Find tilbud ud fra nøgleord
    returliste = FindTilbud(AnalyseList, BrugerID);
  }

  return returliste;
}
